165 string GPIO_PWM::findPWMName(const string path, const string pinName)
166 {
167  DIR* openedDir = opendir(path.c_str());
168  struct dirent* readDir;
169 
170  if (openedDir) {
171 
172  while((readDir = readdir(openedDir)) != 0) {
173  if (strstr(readDir->d_name, pinName.c_str()) != 0) {
174  closedir(openedDir);
175  return readDir->d_name;
176  }
177  }
178 
179  closedir(openedDir);
180  }
181 
182  return "";
